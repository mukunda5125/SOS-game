package com.sosgame.model;

public class SOSGame {
    private final Board board;

    public SOSGame(int size) {
        this.board = new Board(size);
    }

    public Board getBoard() {
        return board;
    }

    /**
     * Place a letter ('S' or 'O') at the given cell if empty.
     * Returns true if move successful, false otherwise.
     */
    public boolean makeMove(int row, int col, char letter) {
        if (letter != 'S' && letter != 'O') return false;
        return board.setCellContent(row, col, letter);
    }
}
